  	showSearchInstructions(true);

    checklistCountryButton.innerHTML = "Country";

    checklistCountryButton.classList.remove("highlight");

      currentChecklistCountryElement.classList.remove("checkHighlight");
      currentChecklistCountry = "";

    if (currentTaxonomyCountry) {
      currentTaxonomyCountryElement.classList.remove("taxHighlight");
      currentTaxonomyCountry = "";

      searchResults.parentNode.parentNode.removeChild(searchResults.parentNode);

      lastQuery           = "";
      // lastTaxonomyCountry = "";

    clearChecklistOptions();
    resetNumDays();
    fillSAMmap();
    currentMap.querySelector(".saveMapButton").style.display = "none";

    setFooterBannerDimensions();
}
