function reloadHomePage(evt)  {

  	showSearchInstructions(true);

    checklistCountryButton.innerHTML = "Country";
    checklistCountryButton.classList.remove("needsAttention");
    checklistCountryButton.classList.remove("highlight");

    taxonomyCountryButton.innerHTML = "Select Country";
    taxonomyCountryButton.classList.remove("needsAttention");
    taxonomyCountryButton.classList.remove("highlight");

    document.querySelector("#searchForm span").classList.add("grayed");
    document.querySelector("#searchSpecials").classList.add("grayed");
    searchInput.value = "";

    if (currentChecklistCountry) {
      currentChecklistCountryElement.classList.remove("checkHighlight");
      currentChecklistCountry = "";
    }

    if (currentTaxonomyCountry) {
      currentTaxonomyCountryElement.classList.remove("taxHighlight");
      currentTaxonomyCountry = "";
      taxTreeArticle.parentNode.removeChild(taxTreeArticle);
      searchResults.parentNode.parentNode.removeChild(searchResults.parentNode);
      map.style.top       = "0";
      lastQuery           = "";
      // lastTaxonomyCountry = "";
      lastResultsSpecies  = "";
    }

    pdfButton.classList.remove("highlight");
    csvButton.classList.remove("highlight");

    clearChecklistOptions();
    resetNumDays();
    fillSAMmap();
    currentMap.querySelector(".saveMapButton").style.display = "none";

    setFooterBannerDimensions();
}

  printWindow.document.close();
  printWindow.print();
});
